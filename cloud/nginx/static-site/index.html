    sendfile           on;
    sendfile_max_chunk 1m;
    tcp_nopush on;

    gzip_types
        application/xml
        text/plain;

        try_files $uri $uri/ $uri.html =404;

    }

    location /public/ { autoindex on; }

    error_page 404 /404.html;
    location = /404.html { internal; }
