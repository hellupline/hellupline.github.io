proxy_cache_key "$scheme$request_method$host$request_uri";

geo $purge_allowed {
   192.168.0.0/24 1;
   127.0.0.1 1;

   default 0;
}

map $request_method $purge_method {
    PURGE $purge_allowed;
    default 0;
}

    server_name _;
    listen [::]:80 default_server ipv6only=on;

    access_log /var/log/nginx/access.log main;

        proxy_cache my_zone;
