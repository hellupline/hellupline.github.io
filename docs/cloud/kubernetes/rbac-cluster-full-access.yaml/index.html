    name: my-cluster-role--full-access

  - apiGroups: ["", "extensions", "apps", "batch"]

    name: my-service-account

    name: my-cluster-role-binding--full-access

    apiGroup: rbac.authorization.k8s.io
    kind: ClusterRole
    name: my-cluster-role--full-access

    kind: ServiceAccount
    name: my-service-account
    namespace: my-namespace

    kind: User
    name: my-user
